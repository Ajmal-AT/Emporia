CREATE TABLE IF NOT EXISTS public.departments (
    id BIGINT NOT NULL GENERATED BY DEFAULT AS IDENTITY (
        INCREMENT 1 START 1 MINVALUE 1 MAXVALUE 9223372036854775807 CACHE 1
    ),
    created_at TIMESTAMP WITHOUT TIME ZONE,
    department_head_id VARCHAR(255) COLLATE pg_catalog."default",
    department_id VARCHAR(255) COLLATE pg_catalog."default" NOT NULL,
    department_name VARCHAR(255) COLLATE pg_catalog."default" NOT NULL,
    description VARCHAR(255) COLLATE pg_catalog."default",
    updated_at TIMESTAMP WITHOUT TIME ZONE,
    CONSTRAINT departments_pkey PRIMARY KEY (id),
    CONSTRAINT uk_department_id UNIQUE (department_id),
    CONSTRAINT uk_department_name UNIQUE (department_name)
) TABLESPACE pg_default;

-- Create Employee Table
CREATE TABLE IF NOT EXISTS public.employees (
    id BIGINT NOT NULL GENERATED BY DEFAULT AS IDENTITY (
        INCREMENT 1 START 1 MINVALUE 1 MAXVALUE 9223372036854775807 CACHE 1
    ),
    created_at TIMESTAMP WITHOUT TIME ZONE,
    date_of_birth DATE NOT NULL,
    date_of_joining DATE NOT NULL,
    department_id VARCHAR(255) COLLATE pg_catalog."default",
    email VARCHAR(255) COLLATE pg_catalog."default" NOT NULL,
    employee_id VARCHAR(255) COLLATE pg_catalog."default" NOT NULL,
    first_name VARCHAR(255) COLLATE pg_catalog."default" NOT NULL,
    gender VARCHAR(255) COLLATE pg_catalog."default" NOT NULL,
    last_name VARCHAR(255) COLLATE pg_catalog."default",
    phone_number VARCHAR(255) COLLATE pg_catalog."default" NOT NULL,
    role VARCHAR(255) COLLATE pg_catalog."default" NOT NULL,
    salary NUMERIC(19,2) NOT NULL,
    updated_at TIMESTAMP WITHOUT TIME ZONE,
    yearly_bonus_percentage DOUBLE PRECISION,
    CONSTRAINT employees_pkey PRIMARY KEY (id),
    CONSTRAINT uk_phone_number UNIQUE (phone_number),
    CONSTRAINT uk_email UNIQUE (email),
    CONSTRAINT uk_employee_id UNIQUE (employee_id)
) TABLESPACE pg_default;

    OWNER TO postgres;
